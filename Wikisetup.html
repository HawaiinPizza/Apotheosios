(setq org-publish-project-alist
        '(

          ("site"
           ;; Hard paths
           :base-directory "/home/zaki/Notes/Apoth/"
           :publishing-directory "/home/zaki/Notes/Apoth/.Site/"
           :base-extension "org"

           ; Sitemap/html
           :sitemap-filename "Main.org"
           :sitemap-title "Apotheosios"
           :auto-sitemap t
           :html-head  "<link rel='stylesheet' type='text/css' href='https://www.w3schools.com/w3css/4/w3.css' />"
           :recursive t

           ; Backend things
           :publishing-function org-html-publish-to-html
           :headline-levels 4
           :html-extension "html"
           :sitemap-sort-files "chronologically"
           :makeindex tex



      )


      ("media"
            :base-directory "/home/zaki/Notes/Apoth/"
            :base-extension "css\\|js\\|png\\|jpg\\|gif\\|pdf\\|mp3\\|ogg\\|swf\\|php"
            :publishing-directory "/home/zaki/Notes/Apoth/.Site/Media"
            :recursive t
            :publishing-function org-publish-attachment)

      ("apoth" :components ("site" "media"))
  ))
